172 bool CMulticlassLogisticRegression::train_machine(CFeatures* data)
173 {
174  if (data)
175  set_features((CDotFeatures*)data);
176 
177  REQUIRE(m_features, "%s::train_machine(): No features attached!\n");
178  REQUIRE(m_labels, "%s::train_machine(): No labels attached!\n");
179  REQUIRE(m_labels->get_label_type()==LT_MULTICLASS, "%s::train_machine(): "
180  "Attached labels are no multiclass labels\n");
181  REQUIRE(m_multiclass_strategy, "%s::train_machine(): No multiclass strategy"
182  " attached!\n");
183 
184  int32_t n_classes = ((CMulticlassLabels*)m_labels)->get_num_classes();
185  int32_t n_feats = m_features->get_dim_feature_space();
186 
187  slep_options options = slep_options::default_options();
188  if (m_machines->get_num_elements()!=0)
189  {
190  SGMatrix<float64_t> all_w_old(n_feats, n_classes);
191  SGVector<float64_t> all_c_old(n_classes);
192  for (int32_t i=0; i<n_classes; i++)
193  {
194  CLinearMachine* machine = (CLinearMachine*)m_machines->get_element(i);
195  SGVector<float64_t> w = machine->get_w();
196  for (int32_t j=0; j<n_feats; j++)
197  all_w_old(j,i) = w[j];
198  all_c_old[i] = machine->get_bias();
199  SG_UNREF(machine);
200  }
201  options.last_result = new slep_result_t(all_w_old,all_c_old);
202  m_machines->reset_array();
203  }
204  options.tolerance = m_epsilon;
205  options.max_iter = m_max_iter;
206  slep_result_t result = slep_mc_plain_lr(m_features,(CMulticlassLabels*)m_labels,m_z,options);
207 
208  SGMatrix<float64_t> all_w = result.w;
209  SGVector<float64_t> all_c = result.c;
210  for (int32_t i=0; i<n_classes; i++)
211  {
212  SGVector<float64_t> w(n_feats);
213  for (int32_t j=0; j<n_feats; j++)
214  w[j] = all_w(j,i);
215  float64_t c = all_c[i];
216  CLinearMachine* machine = new CLinearMachine();
217  machine->set_w(w);
218  machine->set_bias(c);
219  m_machines->push_back(machine);
220  }
221  return true;
222 }
