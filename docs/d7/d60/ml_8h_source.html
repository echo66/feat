165  vector<double> ML::get_weights()
166  {
170  vector<double> w;
171 
172  if (!ml_type.compare("LeastAngleRegression") || !ml_type.compare("LinearRidgeRegression")||
173  !ml_type.compare("SVM") || (!ml_type.compare("LR")))
174  {
175  if(prob_type == PT_MULTICLASS && ( !ml_type.compare("LR")) ) {
176  auto weights = dynamic_pointer_cast<sh::CMulticlassLogisticRegression>(p_est)->get_w();
177 
178 
179  for( int j = 0;j<weights[0].size(); j++)
180  w.push_back(0);
181 
182  for( int i = 0 ; i < weights.size(); i++ ){
183  for( int j = 0;j<weights[i].size(); j++) {
184  w[j] += fabs(weights[i][j]);
185  }
186  }
187 
188  for( int i = 0; i < weights.size() ; i++)
189  w[i] = w[i]/weights.size();;
190  return w;
191  }
192 
193  auto tmp = dynamic_pointer_cast<sh::CLinearMachine>(p_est)->get_w();
194 
195  w.assign(tmp.data(), tmp.data()+tmp.size());
196  for (unsigned i =0; i<w.size(); ++i) // take absolute value of weights
197  w[i] = fabs(w[i]);
198  }
199  else if (!ml_type.compare("CART"))
200  w = dynamic_pointer_cast<sh::CMyCARTree>(p_est)->feature_importances();
201  else
202  {
203  std::cerr << "ERROR: ML::get_weights not implemented for " + ml_type << "\n";
204 
205  }
206 
207  return w;

210  VectorXd ML::fit(MatrixXd& X, VectorXd& y, const Parameters& params, bool& pass,
211  const vector<char>& dtypes)
212  {
228  // for random forest we need to set the number of features per bag
229 
230  init();

236  dynamic_pointer_cast<sh::CRandomForest>(p_est)->set_num_random_features(max_feats);
237  }

241  //std::cout << "setting dtypes\n";

254  //std::cout << "setting labels (n_classes = " << params.n_classes << ")\n";
255  //cout << "y is " << y.transpose() << "\n";

282  y_pred = clf->get_labels();
